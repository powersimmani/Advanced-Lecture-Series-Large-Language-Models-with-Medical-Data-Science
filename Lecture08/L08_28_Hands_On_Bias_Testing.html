from sklearn.metrics import confusion_matrix import pandas as pd # Evaluate performance by group def evaluate_fairness(y_true, y_pred, sensitive_attr): results = {} for group in sensitive_attr.unique(): mask = sensitive_attr == group tn, fp, fn, tp = confusion_matrix( y_true[mask], y_pred[mask] ).ravel() results[group] = { 'TPR': tp / (tp + fn), # Recall 'FPR': fp / (fp + tn), 'PPV': tp / (tp + fp) # Precision } return pd.DataFrame(results).T
